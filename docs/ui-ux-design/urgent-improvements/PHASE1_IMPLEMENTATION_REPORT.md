// Cmd+K ショートカット
useEffect(() => {
  const handleKeyDown = (e: KeyboardEvent) => {
    if ((e.metaKey || e.ctrlKey) && e.key === 'k') {
      e.preventDefault()
      setIsModalOpen(true)
    }
  }
}, [])

// 検索実行
const handleSearch = (searchQuery: string = query) => {
  if (searchQuery.trim()) {
    navigate(`/search?q=${encodeURIComponent(searchQuery.trim())}`)
  }
}
